% Section header - highlighted background with automatic numbering
\newcommand{\sectiontitle}[1]{%
    \stepcounter{sectioncounter}%
    \setcounter{subsectioncounter}{0}%
    \vspace{3pt}%
    \noindent\colorbox{currentsectionbg}{\parbox{\dimexpr\linewidth-2\fboxsep}{\textbf{\large\textcolor{white}{\thesectioncounter. #1}}}}%
    % \vspace{1pt}%
}

    \vspace{1pt}%
}
